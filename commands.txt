====== Login =======
sudo vim /etc/login.defs
	PASS_MAX_DAYS   99999 --> PASS_MAX_DAYS   30
	PASS_MIN_DAYS   0 --> PASS_MIN_DAYS   2
	PASS_WARN_AGE   7
sudo apt install libpam-pwquality
sudo nano /etc/pam.d/common-password
	retry=3
	minlen=10                   // Minimum length 10
	ucredit=-1 dcredit=-1       // Must contain at least one uppercase and one numeric character
	maxrepeat=3                 // Maximum 3 consecutive characters can be used
	reject_username             // Must not contain the username
	difok=7                     // Requirement of 7 different characters not contained in the old password
	enforce_for_root            // We made the rules valid for root as well
	password        requisite                       pam_pwquality.so retry=3 minlen=10 ucredit=-1 dcredit=-1 maxrepeat=3 reject_username difok=7 enforce_for_root
